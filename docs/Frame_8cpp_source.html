108method_info* Frame::obterMethodNamed(StaticClass *classRuntime, string name, string descriptor) {
109 MethodArea &methodArea = MethodArea::getInstance();
110
111 StaticClass *currClass = classRuntime;
112 method_info *method;
113
114 while (currClass != NULL) {
115 ClassFile *classFile = currClass->getClassFile();
116
117 for (int i = 0; i < classFile->methods_count; i++) {
118 method = &(classFile->methods[i]);
119 string methodName = Utils::getFormattedConstant(classFile->constant_pool, method->name_index);
120 string methodDesc = Utils::getFormattedConstant(classFile->constant_pool, method->descriptor_index);
121
122 if (methodName == name && methodDesc == descriptor) {
123 _classRuntime = currClass;
124 return method;
125 }
126 }
127
128 // procurando o método nas super classes.
129 if (classFile->super_class == 0) {
130 currClass = NULL;
131 } else {
132 string superClassName = Utils::getFormattedConstant(classFile->constant_pool, classFile->super_class);
133 currClass = methodArea.carregarClassNamed(superClassName);
134 }
135 }
136
137 return NULL;
138}

143 _codeAttribute = NULL;

146 for (int i = 0; i < _method.attributes_count; i++) {

149
150 if (Utils::compararUtf8String(attrName, "Code")) {
151 _codeAttribute = &(attr->info.code_info);

154 } else if (Utils::compararUtf8String(attrName, "Exceptions")) {
155 _exceptionsAttribute = &(attr->info.exceptions_info);
156 if (_codeAttribute != NULL)

158 }
159 }
