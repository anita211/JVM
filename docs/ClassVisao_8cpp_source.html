31// Helper function to get access flags string
32static const char* getAccessFlagsString(u2 accessFlags) {
33 static char buffer[256];
34 buffer[0] = '\0';
35
36 if (accessFlags & 0x0001) strcat(buffer, "public ");
37 if (accessFlags & 0x0002) strcat(buffer, "private ");
38 if (accessFlags & 0x0004) strcat(buffer, "protected ");
39 if (accessFlags & 0x0008) strcat(buffer, "static ");
40 if (accessFlags & 0x0010) strcat(buffer, "final ");
41 if (accessFlags & 0x0020) strcat(buffer, "super ");
42 if (accessFlags & 0x0040) strcat(buffer, "volatile ");
43 if (accessFlags & 0x0080) strcat(buffer, "transient ");
44 if (accessFlags & 0x0200) strcat(buffer, "interface ");
45 if (accessFlags & 0x0400) strcat(buffer, "abstract ");
46
47 // Remove trailing space
48 size_t len = strlen(buffer);
49 if (len > 0 && buffer[len - 1] == ' ') {
50 buffer[len - 1] = '\0';
51 }
52
53 return buffer;
54}

183// Function to display bytecode
184static void displayByteCode(const Code_attribute& code, cp_info* constantPool, FILE* output) {
185 u4 code_length = code.code_length;
186 const u1* codeData = code.code;
187
188 fprintf(output, "\t\tBytecode:\n");
189
190 uint32_t lineNumber = 0;
191 u4 i = 0;
192
193 while (i < code_length) {
194 printTabs(output, 3);
195
196 u1 opcode = codeData[i];
197 fprintf(output, "%u: %u ", lineNumber++, i);
198
199 if (opcode < 202) { // Valid instruction range
200 fprintf(output, "%s", INSTRUCTION_NAMES[opcode]);
201 } else {
202 fprintf(output, "unknown_%02x", opcode);
203 }
204
205 // Handle instruction operands with constant pool references
206 if (opcode == 0x12) { // ldc
207 u1 index = codeData[i + 1];
208 fprintf(output, " #%d <%s>", index, Utils::getFormattedConstant(constantPool, index));
209 i += 2;
210 } else if (opcode == 0x13 || opcode == 0x14) { // ldc_w, ldc2_w
211 u2 index = (codeData[i + 1] << 8) | codeData[i + 2];
212 fprintf(output, " #%d <%s>", index, Utils::getFormattedConstant(constantPool, index));
213 i += 3;
214 } else if (opcode == 0x10) { // bipush
215 fprintf(output, " %d", (int8_t)codeData[i + 1]);
216 i += 2;
217 } else if (opcode == 0x11) { // sipush
218 int16_t value = (codeData[i + 1] << 8) | codeData[i + 2];
219 fprintf(output, " %d", value);
220 i += 3;
221 } else if ((opcode >= 0x15 && opcode <= 0x19) || (opcode >= 0x36 && opcode <= 0x3a)) { // loads/stores with index
222 fprintf(output, " %d", codeData[i + 1]);
223 i += 2;
224 } else if (opcode == 0x84) { // iinc
225 fprintf(output, " %d by %d", codeData[i + 1], (int8_t)codeData[i + 2]);
226 i += 3;
227 } else if (opcode >= 0xb2 && opcode <= 0xb8) { // field/method references
228 u2 index = (codeData[i + 1] << 8) | codeData[i + 2];
229 fprintf(output, " #%d <%s>", index, Utils::getFormattedConstant(constantPool, index));
230 i += 3;
231 } else if (opcode == 0xb9) { // invokeinterface
232 u2 index = (codeData[i + 1] << 8) | codeData[i + 2];
233 fprintf(output, " #%d <%s> count %d", index, Utils::getFormattedConstant(constantPool, index), codeData[i + 3]);
234 i += 5;
235 } else if (opcode == 0xbb || opcode == 0xbd || opcode == 0xc0 || opcode == 0xc1) { // new, anewarray, checkcast, instanceof
236 u2 index = (codeData[i + 1] << 8) | codeData[i + 2];
237 fprintf(output, " #%d <%s>", index, Utils::getFormattedConstant(constantPool, index));
238 i += 3;
239 } else if (opcode >= 0x99 && opcode <= 0xa6) { // conditional jumps
240 int16_t offset = (codeData[i + 1] << 8) | codeData[i + 2];
241 fprintf(output, " %u (%+d)", (unsigned)(i + offset), offset);
242 i += 3;
243 } else if (opcode == 0xa7 || opcode == 0xa8) { // goto, jsr
244 int16_t offset = (codeData[i + 1] << 8) | codeData[i + 2];
245 fprintf(output, " %u (%+d)", (unsigned)(i + offset), offset);
246 i += 3;
247 } else {
248 i += 1;
249 }
250
251 fprintf(output, "\n");
252 }
253}

255// Function to display attributes
256static void displayAttribute(const attribute_info& attr, cp_info* constantPool, FILE* output, int indentation) {
257 const char* attributeName = Utils::getFormattedConstant(constantPool, attr.attribute_name_index);
258
259 printTabs(output, indentation);
260 fprintf(output, "Attribute: %s\n", attributeName);
261 printTabs(output, indentation + 1);
262 fprintf(output, "Attribute name index: cp_info #%d\n", attr.attribute_name_index);
263 printTabs(output, indentation + 1);
264 fprintf(output, "Attribute length: %u\n", attr.attribute_length);
265
266 CONSTANT_Utf8_info attributeUtf8 = constantPool[attr.attribute_name_index - 1].info.utf8_info;
267
268 if (Utils::compararUtf8String(attributeUtf8, "Code")) {
269 const Code_attribute& code = attr.info.code_info;
270 printTabs(output, indentation + 1);
271 fprintf(output, "Maximum stack depth: %d\n", code.max_stack);
272 printTabs(output, indentation + 1);
273 fprintf(output, "Maximum local variables: %d\n", code.max_locals);
274 printTabs(output, indentation + 1);
275 fprintf(output, "Code length: %u\n", code.code_length);
276
277 displayByteCode(code, constantPool, output);
278
279 printTabs(output, indentation + 1);
280 fprintf(output, "Exception table length: %d\n", code.exception_table_length);
281
282 if (code.attributes_count > 0) {
283 printTabs(output, indentation + 1);
284 fprintf(output, "Attributes:\n");
285 for (uint16_t i = 0; i < code.attributes_count; i++) {
286 displayAttribute(code.attributes[i], constantPool, output, indentation + 2);
287 }
288 }
289 } else if (Utils::compararUtf8String(attributeUtf8, "SourceFile")) {
290 const SourceFile_attribute& sourceFile = attr.info.sourceFile_info;
291 printTabs(output, indentation + 1);
292 fprintf(output, "Source file: cp_info #%d <%s>\n", sourceFile.sourcefile_index,
293 Utils::getFormattedConstant(constantPool, sourceFile.sourcefile_index));
294 } else if (Utils::compararUtf8String(attributeUtf8, "LineNumberTable")) {
295 const LineNumberTable_attribute& lineNumberTable = attr.info.lineNumberTable_info;
296 printTabs(output, indentation + 1);
297 fprintf(output, "Line number table length: %d\n", lineNumberTable.line_number_table_length);
298 for (uint16_t i = 0; i < lineNumberTable.line_number_table_length; i++) {
299 const LineNumberTable& entry = lineNumberTable.line_number_table[i];
300 printTabs(output, indentation + 2);
301 fprintf(output, "line %d: %d\n", entry.line_number, entry.start_pc);
302 }
303 }
