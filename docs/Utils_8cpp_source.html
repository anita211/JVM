125 CONSTANT_Integer_info intInfo = constant.info.integer_info;
126 int32_t number = intInfo.bytes;

130 } else if (constant.tag == CONSTANT_Float) {
131 CONSTANT_Float_info floatInfo = constant.info.float_info;
132
133 int32_t s = ((floatInfo.bytes >> 31) == 0) ? 1 : -1;
134 int32_t e = ((floatInfo.bytes >> 23) & 0xff);

162
163 const char *name = Utils::getFormattedConstant(constantPool, nameAndTypeInfo.name_index);
164 const char *descriptor = Utils::getFormattedConstant(constantPool, nameAndTypeInfo.descriptor_index);
165
166 char *result = static_cast<char*>(malloc(sizeof(char) * (strlen(name) + strlen(descriptor)) + 1));
167 strcpy(result, name);
168 strcat(result, descriptor);
169
170 return result;
171 } else if (constant.tag == CONSTANT_Utf8) {
172 CONSTANT_Utf8_info utf8_info = constant.info.utf8_info;
173 char *str = static_cast<char*>(malloc((utf8_info.length + 1) * sizeof(char)));
174
175 uint16_t j = 0;
176 for (uint16_t i = 0; i < utf8_info.length; i++) {
177 if (isprint(utf8_info.bytes[i]))
178 str[j++] = utf8_info.bytes[i];
179 }
180 str[j] = '\0';
181
182 return str;
183 } else {
184 cerr << "Arquivo .class possui uma tag " << constant.tag << " invalida no pool de constantes." << endl;
185 exit(5);
186 }
