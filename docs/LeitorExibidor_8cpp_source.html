132void LeitorExibidor::setConstantPool(FILE *fp, ClassFile *classFile) {
133 u2 poolSize = classFile->constant_pool_count - 1;
134 classFile->constant_pool = static_cast<cp_info*>(malloc(sizeof(cp_info) * poolSize));
135
136 cp_info *constant_pool = classFile->constant_pool;
137 for (u2 i = 0; i < poolSize; i++) {
138 u1 tag = readU1(fp);
139 constant_pool[i].tag = tag;
140
141 switch (tag) {
142 case CONSTANT_Class:
143 constant_pool[i].info.class_info = getConstantClassInfo(fp);
144 break;
145 case CONSTANT_Fieldref:
146 constant_pool[i].info.fieldref_info = getConstantFieldRefInfo(fp);
147 break;
148 case CONSTANT_Methodref:
149 constant_pool[i].info.methodref_info = getConstantMethodRefInfo(fp);
150 break;
151 case CONSTANT_InterfaceMethodref:
152 constant_pool[i].info.interfaceMethodref_info = getConstantInterfaceMethodRefInfo(fp);
153 break;
154 case CONSTANT_String:
155 constant_pool[i].info.string_info = getConstantStringInfo(fp);
156 break;
157 case CONSTANT_Integer:
158 constant_pool[i].info.integer_info = getConstantIntegerInfo(fp);
159 break;
160 case CONSTANT_Float:
161 constant_pool[i].info.float_info = getConstantFloatInfo(fp);
162 break;
163 case CONSTANT_Long:
164 constant_pool[i].info.long_info = getConstantLongInfo(fp);
165 constant_pool[++i].tag = CONSTANT_NULL;
166 break;
167 case CONSTANT_Double:
168 constant_pool[i].info.double_info = getConstantDoubleInfo(fp);
169 constant_pool[++i].tag = CONSTANT_NULL;
170 break;
171 case CONSTANT_NameAndType:
172 constant_pool[i].info.nameAndType_info = getConstantNameAndTypeInfo(fp);
173 break;
174 case CONSTANT_Utf8:
175 constant_pool[i].info.utf8_info = getConstantUtf8Info(fp);
176 break;
177 default:
178 cerr << "Arquivo .class possui uma tag invalida no pool de constantes";
179 exit(5);
180 }
181 }
182}

440attribute_info LeitorExibidor::getAttributeInfo(FILE *fp, ClassFile *classFile) {
441 attribute_info result;
442 result.attribute_name_index = readU2(fp);
443 result.attribute_length = readU4(fp);

448 } else if (Utils::compararUtf8String(name, "Code")) {
449 result.info.code_info = getAttributeCode(fp, classFile);

451 result.info.exceptions_info = getAttributeExceptions(fp);
452 } else if (Utils::compararUtf8String(name, "InnerClasses")) {
453 result.info.innerClasses_info = getAttributeInnerClasses(fp);
454 } else if (Utils::compararUtf8String(name, "Synthetic")) {
455 result.info.synthetic_info = getAttributeSynthetic();
456 } else if (Utils::compararUtf8String(name, "SourceFile")) {
457 result.info.sourceFile_info = getAttributeSourceFile(fp);
458 } else if (Utils::compararUtf8String(name, "LineNumberTable")) {
459 result.info.lineNumberTable_info = getAttributeLineNumberTable(fp);
460 } else if (Utils::compararUtf8String(name, "LocalVariableTable")) {
461 result.info.localVariableTable_info = getAttributeLocalVariable(fp);
462 } else if (Utils::compararUtf8String(name, "Deprecated")) {
463 result.info.deprecated_info = getAttributeDeprecated();
464 } else if (Utils::compararUtf8String(name, "StackMapTable")) {
465
466 for (int i = 0; i < (signed) result.attribute_length; i++) {
467 readU1(fp);

469 } else {
